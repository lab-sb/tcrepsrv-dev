        alter table public.prova add ad varchar(20);

$ sh runTWrapperSQL.sh  -n prod1 -s /tmp/sql-script.sql
Logging startup messages to : TWrapperSQL_prod1_2020-01-13-15:09:59.log

prod1
/tmp/sql-script.sql

walq _rdb_bdr.walq__prod1
